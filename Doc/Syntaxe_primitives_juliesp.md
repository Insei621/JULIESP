# Le Cœur (Formes Spéciales) :
- quote (')     -> ²
- if            -> ?
- setq          -> :
- lambda        -> £    // le type de retour d'une fonction lambda est toujours un int
- progn         -> ;
- load          -> $        //ne marchera que sur linux
- print         -> €
- scanf         -> ç   // penser à déclarer la variable avant de l'appeler, stocke des int

# Les Mains (Primitives de Listes) :
- car           -> <<
- cdr           -> >>
- cons          -> &
- null (pour savoir si c'est vide) -> |
- atom (pour vérifier si c'est une liste ou un atome) -> @

# Le Calcul (Primitives Arithmétiques) :
- +, -, *, /
- <, >, =
- adreq        -> ¤
- numberq   -> °

# Booleen
- T (vrai)      -> µ
- NIL (faux)    -> ù

# commentaire en lisp
- ; ceci est un commentaire sur une ligne   -> §§
- #| This is a multi-line comment. |#       -> §!    !§
- . pour les virgule des décimaux
